练习1、图形面积计算系统

图形面积计算系统，创建一个基类 Shape（图形），包含属性：名称(name)，方法：计算面积(calculate_area)，返回0；（1）创建2个子类：Rectangle（矩形）：继承Shape，新增属性长(length)、宽(width)，重写calculate_area方法；Circle（圆形）：继承Shape，新增属性半径(radius)，重写calculate_area方法。（2）根据用户输入的图形类型，选择输出对应的面积，要求有构造函数的使用。

知识点：继承、方法重写、构造函数 __init__
参考代码
class Shape:
    def __init__(self, name):
        self.name = name
    def calculate_area(self):
        return 0

class Rectangle(Shape):
    def __init__(self, length, width):
        super().__init__('矩形')
        self.length = length
        self.width = width
    def calculate_area(self):
        return self.length * self.width

class Circle(Shape):
    def __init__(self, radius):
        super().__init__('圆形')
        self.radius = radius
    def calculate_area(self):
        return 3.14159 * self.radius ** 2

# 使用示例
choice = input('请输入图形类型(矩形/圆形): ')
if choice == '矩形':
    l = float(input('长: '))
    w = float(input('宽: '))
    shape = Rectangle(l, w)
else:
    r = float(input('半径: '))
    shape = Circle(r)
print(f'面积: {shape.calculate_area()}')

练习2、猜数字游戏

猜数字游戏：一个类A有一个成员变量v，有一个初值100。定义一个类，对A类的成员变量v进行猜。如果大了则提示大了，小了则提示小了。等于则提示猜测成功。

参考代码
class A:
    def __init__(self):
        self.v = 100

class GuessGame:
    def __init__(self, target):
        self.target = target
    def guess(self, num):
        if num > self.target.v:
            print('大了')
        elif num < self.target.v:
            print('小了')
        else:
            print('猜测成功!')

a = A()
game = GuessGame(a)
while True:
    n = int(input('请输入猜测的数: '))
    game.guess(n)
    if n == a.v:
        break

练习6、Person类冒泡排序

定义一个Person类（有属性：name, age, job），初始化Person对象数组，有3个person对象，并按照age从大到小进行排序后输出名字和对应的年龄，要求使用冒泡排序。

知识点：冒泡排序、对象比较、__str__ 方法
参考代码
class Person:
    def __init__(self, name, age, job):
        self.name = name
        self.age = age
        self.job = job

persons = [
    Person('张三', 25, '工程师'),
    Person('李四', 30, '教师'),
    Person('王五', 22, '学生')
]

# 冒泡排序（按age从大到小）
for i in range(len(persons)):
    for j in range(len(persons)-1-i):
        if persons[j].age < persons[j+1].age:
            persons[j], persons[j+1] = persons[j+1], persons[j]

for p in persons:
    print(f'姓名:{p.name} 年龄:{p.age}')

class Employee:
    def __init__(self, name, age, post, salary):
        self.name = name
        self.age = age
        self.post = post
        self.salary = salary
    def introduce(self):
        print(f'姓名:{self.name} 年龄:{self.age} '
              f'职称:{self.post} 工资:{self.salary}')

class Manager(Employee):
    def __init__(self, name, age):
        super().__init__(name, age, '经理', 0)
        self.salary = 3.3 * 5000  # 基本工资*系数
    def introduce(self):
        print(f'[经理] {self.name} {self.age}岁 '
              f'工资:{self.salary:.1f}')

class Director(Employee):
    def __init__(self, name, age):
        super().__init__(name, age, '主任', 0)
        self.salary = 2.1 * 5000
    def introduce(self):
        print(f'[主任] {self.name} {self.age}岁 '
              f'工资:{self.salary:.1f}')

class Staff(Employee):
    def __init__(self, name, age):
        super().__init__(name, age, '员工', 0)
        self.salary = 1.0 * 5000
    def introduce(self):
        print(f'[员工] {self.name} {self.age}岁 '
              f'工资:{self.salary:.1f}')

# 测试

练习3、异常值识别（3σ原则）

读懂下列代码后，充分了解其功能后，在程序中实现。

all_info_notnull.csv
3σ原则：数据落在均值 ± 3倍标准差 之外视为异常值。公式：|x - mean| > 3 × std 
all_info = pd.read_csv('../tmp/all_info_notnull.csv')
# 定义3σ原则函数来识别异常值
def out_range(ser1):
    bool_ind = (ser1.mean() - 3 * ser1.std() > ser1) |  (ser1.mean() + 3 * ser1.var() < ser1)
    index = np.arange(ser1.shape[0])[bool_ind]
    outrange = ser1.iloc[index]
    return outrange

outlier = out_range(all_info['年龄/岁'])
print('使用3σ原则判定异常值个数为: ', outlier.shape[0])
print('异常值的最大值为: ', outlier.max())
print('异常值的最小值为: ', outlier.min())

练习5、三种数据标准化方法

读懂下列代码后，充分了解其功能后，在程序中实现。

user_pay_info.csv
三种标准化对比：
离差标准化：(x - min) / (max - min)，结果在 [0,1] 之间
标准差标准化：(x - mean) / std，均值为0，标准差为1
小数定标标准化：x / 10^ceil(log10(|x|max))，按数量级归一化 
import pandas as pd
pay = pd.read_csv('../data/user_pay_info.csv', index_col=0)

# 自定义离差标准化函数
def min_max_scale(data):
    data = (data - data.min()) / (data.max() - data.min())
    return data

# 对用户每月支出信息表的每月支出数据进行离差标准化
pay_min_max = min_max_scale(pay['每月支出/元'])
print('离差标准化之前每月支出数据为: \n', pay['每月支出/元'].head())
print('离差标准化之后每月支出数据为: \n', pay_min_max.head())

# 自定义标准差标准化函数
def standard_scaler(data):
    data = (data - data.mean()) / data.std()
    return data

# 对用户每月支出信息表的每月支出数据进行标准差标准化
pay_standard = standard_scaler(pay['每月支出/元'])
print('标准差标准化之前每月支出数据为: \n', pay['每月支出/元'].head())
print('标准差标准化之后每月支出数据为: \n', pay_standard.head())

# 自定义小数定标标准化函数
import numpy as np
def decimal_scaler(data):
    data = data / 10 ** np.ceil(np.log10(data.abs().max()))
    return data

# 对用户每月支出信息表的每月支出数据进行小数定标标准化
pay_decimal = decimal_scaler(pay['每月支出/元'])
print('小数定标标准化之前的每月支出数据为: \n', pay['每月支出/元'].head())
print('小数定标标准化之后的每月支出数据为: \n', pay_decimal.head())

练习6、哑变量处理 + 数据离散化

读懂下列代码后，充分了解其功能后，在程序中实现。

user_all_info.csv
三种离散化方法：
pd.cut() — 等宽法，按数值范围等分
自定义等频法 — 按分位数切分，每箱个数相同
K-Means聚类 — 用聚类算法自动找分界点 
import pandas as pd
import numpy as np
all_info = pd.read_csv('../data/user_all_info.csv')
live_type = all_info.loc[0: 5, '居住类型']  # 抽取部分数据做演示
print('哑变量处理前的数据为: \n', live_type)
print('哑变量处理后的数据为: \n', pd.get_dummies(live_type, dtype=int))

age_cut = pd.cut(all_info['年龄/岁'], 5)
print('离散化后记录的年龄分布为: \n', age_cut.value_counts())

# 自定义等频法离散化函数
def same_rate_cut(data, k):
    h = data.quantile(np.arange(0, 1 + 1.0 / k, 1.0 / k))
    data = pd.cut(data, h)
    return data

# 对用户年龄进行等频法离散化
age_same_rate = same_rate_cut(all_info['年龄/岁'], 5).value_counts()
print('用户年龄数据等频法离散化后分布状况为: ', '\n', age_same_rate)

# K-Means聚类法离散化
def kmean_cut(data, k):
    from sklearn.cluster import KMeans  # 引入K-Means
    # 建立模型
    kmodel = KMeans(n_clusters=k, random_state=0)
    kmodel.fit(data.values.reshape((len(data), 1)))  # 训练模型
    # 输出聚类中心并排序
    c = pd.DataFrame(kmodel.cluster_centers_).sort_values(0)
    w = c.rolling(2).mean().iloc[1:]  # 对相邻两端求中点，作为边界点
    w = [0] + list(w[0]) + [data.max()]  # 把首末边界点加上
    data = pd.cut(data, w)
    return data

# 用户年龄聚类分析法离散化
all_info.dropna(inplace=True)
age_kmeans = kmean_cut(all_info['年龄/岁'], 5).value_counts()
print('用户年龄聚类分析法离散化后各个类别数目分布状况为: ', '\n', age_kmeans)
